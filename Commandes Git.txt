git status //Permet de visualiser les fichiers qui ont été modifiés
//en vert ceux add au commit, en rouge ce a add (ou non) au commit

//Pour add un fichier au commit (si le nom du fichier contient un espace le mettre entre "" )
git add nomDuFichier.extension
//Pour suprimer les modifications effectués sur un fichier (erreurs, souhait de revenir à la version précédente, ...)
git checkout -- nomDuFichier.extension

//Une fois les actions ci-dessus effectuées
git commit -m "Message descriptif du commit"

//On charge les potentielles modifications effectuées sur le git par un collaborateur
git pull --rebase

//!!!!!//
Lors du pull si un collaborateur à modifié un fichier qui à été add au commit un conflit aura lieu (d'ou la nécéssité de se repartir les tâches et de toujours tenir ses collaborateurs au courant des fichiers sur lesquels ont travail) ---> suivre les indications du terminal
//!!!!!//

//On push nos modif
git push origin master //master est le nom de la branche, si on souhaite push sur une branche "toto" on fera un : git push origin toto
